快速读入头

#include<bits/stdc++.h>
#define ll long long
using namespace std;
void read(ll& x)
{
    int f = 1;
    x = 0;
    char ch = getchar();
    
    while (ch < '0' || ch > '9')
    {
        if (ch == '-')
            f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9')
    {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    x *= f;
}
