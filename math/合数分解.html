大范围少调用

//这个板子只能处理正数

//合数分解O（sqrt(n)）
//最大使用范围[1,maxn),实际使用[1,maxn);
//初始化要求prime空,no_prime为0，调用find_prime()即可
const ll maxn=1e5+5;
bool no_prime[maxn];
ll prime[maxn],prime_;

void find_prime() //find prime in [2,maxn];
{
    no_prime[1]=true;
    ll n=maxn;
    for(ll i=2; i<n; i++)
    {
        if(!no_prime[i])
        {
            prime[prime_++]=i;
        }
        for(ll j=0; j<prime_&&prime[j]*i<n; j++)
        {
            no_prime[prime[j]*i]=true;
            if(i%prime[j]==0)
                break;
        }
    }
}
ll fac[100][2],fac_;
//当x=0，1会异常，
void getfac(ll x)
{
    fac_=0;
    for(ll i=0; prime[i]<=x/prime[i]; i++)
    {
        if(x%prime[i]==0)
        {
            fac[fac_][0]=prime[i];
            fac[fac_][1]=0;
            while(x%prime[i]==0)
            {
                fac[fac_][1]++;
                x/=prime[i];
            }
            fac_++;
        }
    }
    if(x!=1)
    {
        fac[fac_][0]=x;
        fac[fac_][1]=1;
        fac_++;
    }
}

小范围多调用

//无法处理负数

//预处理O（n）合数分解O（lgn）
//最大使用范围[1,maxn),实际使用[1,maxn);
//初始化要求prime空,no_prime为0，调用find_prime()即可
const int maxn=1.5e7+5;
bool no_prime[maxn];
int minfac[maxn];//预处理之后可以O（lgn）分解
int prime[maxn/10],prime_;

void find_prime() //find prime in [2,maxn];
{
    minfac[1]=1;
    minfac[2]=2;
    no_prime[1]=true;
    int n=maxn;
    for(int i=2; i<n; i++)
    {
        if(!no_prime[i])
        {
            prime[prime_++]=i;
            minfac[i]=i;
        }
        for(int j=0; j<prime_&&prime[j]*i<n; j++)
        {
            no_prime[prime[j]*i]=true;
            minfac[prime[j]*i]=prime[j];
            if(i%prime[j]==0)
                break;
        }
    }
}
int fac[100][2],fac_;
//当x=0，1会异常，
void getfac(int x)
{
    fac_=0;
    while(x!=1)
    {
        int little=minfac[x];
        fac[fac_][0]=little;
        fac[fac_][1]=0;
        while(little!=1&&minfac[x]==little)
        {
            x/=little;
            fac[fac_][1]++;
        }
        fac_++;
    }
}
