AC自动机

struct trie{
    static const int maxn=4e6+5;
    int nex[maxn][26],fail[maxn],end[maxn];
    int flag[maxn],dep[maxn];
    int root,cnt;
    
    void ini(){
        memset(flag,0,sizeof(flag));
        cnt=0;
        //////////////////
        for(int i=0;i<26;i++)nex[cnt][i]=-1;
        dep[cnt]=1;
        end[cnt]=0;
        fail[cnt]=-1;
        //////////////////
        root=cnt++;
    }
    void insert(char buf[]){
        int len=(int)strlen(buf);
        int now=root;
        for(int i=0;i<len;i++){
            if(nex[now][buf[i]-'a']==-1){
                ///////////////////////
                for(int i=0;i<26;i++)nex[cnt][i]=-1;
                dep[cnt]=dep[now]+1;
                end[cnt]=0;
                fail[cnt]=-1;
                //////////////////////
                nex[now][buf[i]-'a']=cnt++;
            }
            now=nex[now][buf[i]-'a'];
        }
        end[now]++;
    }
    void get_fail(){
        queue<int>q;
        fail[root]=root;
        for(int i=0;i<26;i++){
            if(nex[root][i]==-1)
                nex[root][i]=root;//如果不存在这个儿子，那么直接去找根
            else{
                fail[nex[root][i]]=root;//如果存在这个儿子，那么失配时去找根
                q.push(nex[root][i]);
            }
        }
        while(!q.empty()){
            int now=q.front();
            q.pop();
            for(int i=0;i<26;i++){
                if(nex[now][i]==-1)
                    nex[now][i]=nex[fail[now]][i];
                else{
                    fail[nex[now][i]]=nex[fail[now]][i];
                    q.push(nex[now][i]);
                }
            }
        }
    }
    int query(char buf[]){
        int len=(int)strlen(buf);
        int now=root;
        int res=0;
        for(int i=0;i<len;i++){
            char s=buf[i];
            if(s>='a'&&s<='z')s-='a';
            else if(s>='A'&&s<='Z')s-='A';
            else {
                now=root;
                continue;
            }
            
            now=nex[now][s];
            int tmp=now;
            while(tmp!=root){
                res+=end[tmp];
                //end[tmp]=0;
                if(end[tmp]!=0){
                    flag[i-dep[tmp]+2]--;
                    flag[i+1]++;
                }
                tmp=fail[tmp];
            }
        }
        return res;
    }
    void debug(){
        for(int i=0;i<35;i++)printf(" ");
        for(int j=0;j<26;j++){
            printf("%2c",j+'a');
        }
        puts("");
        for(int i=0;i<cnt;i++){
            printf("id=%3d | fail=%3d | end=%3d | chi=[",i,fail[i],end[i]);
            for(int j=0;j<26;j++){
                printf("%2d",nex[i][j]);
            }
            printf("]\n");
        }
    }
}ac;
