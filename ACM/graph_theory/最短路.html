//最短/长路spfa算法
//可以用于解决差分约束问题
//对于同一张图可以多次调用spfa()，你不需要担心初始化问题
//对图进行修改/初始化时要调用ini()

//关于差分约束问题
//如果是大于号不等式x1>=x2+k,则连边x2->x1,对应找最长路
//如果是小于号不等式x1<=x2+k,则连边x2->x1,对应找最短路

struct sofc{
    static const int maxn=1e5+5;
    static const int maxm=3e5+5;
    int head[maxn],cir[maxn],d[maxn];
    bool inq[maxn];
    int n,cnt;
    struct edge{int v,w,nex;}g[maxm];
    
    void ini(int newn){
        n=newn;
        cnt=0;
        memset(head,-1,sizeof(head));
    }
    void add_edge(int u,int v,int w){
        g[cnt].v=v;
        g[cnt].w=w;
        g[cnt].nex=head[u];
        head[u]=cnt++;
    }
    bool spfa(int s){
        //fill(d,d+n+1,1e9);//最短路选这个
        memset(d,0,sizeof(d));//最长路选这个
        memset(cir,0,sizeof(cir));
        memset(inq,0,sizeof(inq));
    
        queue<int>q;
        q.push(s);
        inq[s]=true;
        d[s]=0;cir[s]=1;
        
        while(!q.empty()){
            int u=q.front();q.pop();
            inq[u]=false;
            for(int i=head[u];~i;i=g[i].nex){
                int v=g[i].v,w=g[i].w;
                if(d[v]<d[u]+w){//大于号就是求最短路，小于号就是求最长路
                    d[v]=d[u]+w;
                    if(++cir[v]>=n)return false;
                    if(!inq[v]){
                        inq[v]=1;
                        q.push(v);
                    }
                }
            }
        }
        return true;
    }
}g;
