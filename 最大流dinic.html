最大流dinic算法

//最大使用范围[0,N][0,M],INF是流量达不到的无穷大
//add必须成对使用,注意M并不是简单的边数
//初始化要求cnt=0,head;
const int N,M,INF;
int cnt;
struct edge{int u,v,c,nex;}g[M+1];
int lv[N+1],current[N+1],head[N+1];
void add(int u,int v,int c){
	g[cnt].u=u,g[cnt].v=v,g[cnt].c=c;
	g[cnt].nex=head[u],head[u]=cnt++;
}
void bfs(int s){
	memset(lv,-1,sizeof(lv));
	lv[s]=0;
	queue<int>q;
	q.push(s);
	while(!q.empty()){
		int u=q.front();q.pop();
		for(int i=head[u];i+1;i=g[i].nex){
			edge&e=g[i];
			if(e.c<=0||lv[e.v]>=0)continue;
			lv[e.v]=lv[u]+1;
			q.push(e.v);
		}
	}
}
int dfs(int u,int t,int f){
	if(u==t)return f;
	for(int&i=current[u];i+1;i=g[i].nex){
		edge&e=g[i],&rev=g[i^1];
		if(e.c<=0||lv[u]>=lv[e.v])continue;
		int d=dfs(e.v,t,min(f,e.c));
		if(d<=0)continue;
		e.c  -=d;
		rev.c+=d;
		return d;
	}
	return 0;
}
int dinic(int s,int t){
	int flow=0;
	while(true){
		memmove(current,head,sizeof(head));
		bfs(s);
		if(lv[t]<0)return flow;
		int f;
		while((f=dfs(s,t,INF))>0)flow+=f;//注意括号
	}
}
//至于最大权闭合图,将正权点连接s,容量为点权,负权点连接t容量为点权绝对值,所求得的最小割将图分为两部分,我们发现所有正点权的和减去该简单割便是与s相连的闭合图的权，
通过最小割找到了最大权闭合图。
