多项式求逆

const int maxn=8e5+5,mod = 998244353,g=3;// const 能明显加速代码
int qpow(int a,int b){
    int ret=1;
    while(b){
        if(b&1) ret=1ll*ret*a%mod;
        a=1ll*a*a%mod;
        b>>=1;
    }
    return ret;
}

/////////////////// ntt
int r[maxn],lstn;
void ntt(int*a,int n,int bits,int opt){
    if(lstn!=n) for(int i=0;i<n;i++) r[i]=(r[i>>1]>>1)|((i&1)<<(bits-1));
    for(int i=0;i<n;i++) if(i<r[i]) swap(a[i],a[r[i]]);
    lstn=n;

    for(int k=1;k<n;k<<=1){
        int wn=qpow(g,(mod-1)/(k<<1));//用原根代替单位根
        if(opt==-1) wn=qpow(wn,mod-2);//逆变换则乘上逆元
        for (int i=0;i<n;i+=(k<<1)){
            for (int j=0,w=1; j<k; j++,w=1ll*w*wn%mod){
                int x=a[i+j], y=1ll*w*a[i+j+k]%mod;
                a[i+j]=(x+y)%mod, a[i+j+k]=(x-y+mod)%mod;
            }
        }
    }

    if(opt==-1) {
        int inv=qpow(n,mod-2);
        for(int i=0;i<n;i++) a[i]=1ll*a[i]*inv%mod;
    }
}

int cpa[maxn];
void poly_inv(int*a,int n,int*ret){ //%(x^n)
    if(n==1) {ret[0]=ret[0]=qpow(a[0],mod-2); return;}
    poly_inv(a,(n+1)>>1,ret);

    int tmp=1,bits=0;// 项数 = deg+1
    while(tmp<n*2) tmp<<=1, bits++;

    for(int i=0;i<n;i++) cpa[i]=a[i];
    for(int i=n;i<tmp;i++) cpa[i]=0;

    ntt(cpa,tmp,bits,1);
    ntt(ret,tmp,bits,1);
    for(int i=0;i<tmp;i++) ret[i]=ret[i]*(2-1ll*cpa[i]*ret[i]%mod+mod)%mod;
    ntt(ret,tmp,bits,-1);
    for(int i=n;i<tmp;i++) ret[i]=0;
}
