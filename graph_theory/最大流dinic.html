const double eps=1e-12;
struct dinic
{
    static const int maxn=1e5;
    static const int maxm=2e6;
    static const int inf=1e9;
    
    int cnt;
    struct edge
    {
        int v,nex;
        double c;
    } g[2*maxm];
    int lv[maxn],current[maxn],head[maxn];
    void add_edge(int u,int v,double c)
    {
        g[cnt].v=v;
        g[cnt].c=c;
        g[cnt].nex=head[u];
        head[u]=cnt++;
        
        g[cnt].v=u;
        g[cnt].c=0;
        g[cnt].nex=head[v];
        head[v]=cnt++;
    }
    void ini()
    {
        memset(head,-1,sizeof(head));
        cnt=0;
    }
    void bfs(int s)
    {
        memset(lv,-1,sizeof(lv));
        lv[s]=0;
        queue<int>q;
        q.push(s);
        while(!q.empty())
        {
            int u=q.front();
            q.pop();
            for(int i=head[u]; ~i; i=g[i].nex)
            {
                edge&e=g[i];
                if(e.c<2*eps||lv[e.v]>=0)
                    continue;
                lv[e.v]=lv[u]+1;
                q.push(e.v);
            }
        }
    }
    double dfs(int u,int t,double f)
    {
        if(u==t)
            return f;
        for(int&i=current[u]; ~i; i=g[i].nex)
        {
            edge&e=g[i],&rev=g[i^1];
            if(e.c<eps||lv[u]>=lv[e.v])
                continue;
            double d=dfs(e.v,t,min(f,e.c));
            if(d<eps)
                continue;
            e.c  -=d;
            rev.c+=d;
            return d;
        }
        return 0;
    }
    double maxflow(int s,int t)
    {
        double flow=0;
        while(true)
        {
            memmove(current,head,sizeof(head));
            bfs(s);
            if(lv[t]<0)
                return flow;
            double f;
            while((f=dfs(s,t,inf))>eps)
                flow+=f;//注意括号
        }
    }
} g;

//我们发现所有正点权的和减去该简单割便是与s相连的闭合图的权，通过最小割找到了最大权闭合图。
