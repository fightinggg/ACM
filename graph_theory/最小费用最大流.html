最小费用最大流

struct MCMF
{
    static const int maxn = 50100;
    static const int maxm = 5001000;
    static const int INF = 0x7FFFFFFF;
    static const int INF0X3F = 0x3f3f3f3f;
    int n, m, head[maxn], s, t, cnt;
    int dist[maxn], inq[maxn], pre[maxn], incf[maxn];
    int max_flow, min_cost;
    
    struct Edge
    {
        int to, c, w, next;
    } g[maxm * 4];
    
    void init(int l, int r, int ss, int tt)
    {
        //for(int i = l; i <= r; i++ ){head[i] = -1;}
        memset(head, -1, sizeof(head));
        s = ss;
        t = tt;
        cnt = 0;
        max_flow = min_cost = 0;
    }
    
    void add_g(int u, int v, int c, int w)
    {
        g[cnt].to = v;
        g[cnt].c = c;
        g[cnt].w = w;
        g[cnt].next = head[u];
        head[u] = cnt++;
        
        g[cnt].to = u;
        g[cnt].c = 0;
        g[cnt].w = -w;
        g[cnt].next = head[v];
        head[v] = cnt++;
    }
    
    bool spfa(int s, int t)
    {
        for(int i = 0; i < maxn; i++ )
        {
            dist[i] = INF;
            inq[i] = 0;
        }
        queue<int>que;
        que.push(s);
        inq[s] = 1;
        dist[s] = 0;
        incf[s] = INF0X3F;
        while(!que.empty())
        {
            int u = que.front();
            que.pop();
            inq[u] = 0;
            for(int i = head[u]; ~i; i = g[i].next)
            {
                int to = g[i].to, c = g[i].c, w = g[i].w;
                if(c > 0 && dist[to] > dist[u] + w)
                {
                    dist[to] = dist[u] + w;
                    incf[to] = min(incf[u], c);
                    pre[to] = i;
                    if(!inq[to])
                    {
                        que.push(to);
                        inq[to] = 1;
                    }
                }
            }
        }
        return dist[t] != INF;
    }
    void update(int s, int t)
    {
        int x = t;
        while(x != s)
        {
            int pos = pre[x];
            g[pos].c -= incf[t];
            g[pos ^ 1].c += incf[t];
            x = g[pos ^ 1].to;
        }
        max_flow += incf[t];
        min_cost += dist[t] * incf[t];
    }
    void minCostMaxFlow(int s, int t)
    {
        while(spfa(s, t))
        {
            update(s, t);
        }
    }
} p;
