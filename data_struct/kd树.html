kdTree

       kd树是平衡树的多维拓展，说白了就是多维平衡树，它 和普通的的区别就在于，它是按照深度决定以哪个维度 作为建树划分标准的。（优化算法另当别论）
         算法在注释里面已经很清楚了。 
//kd树，占用了三个变量名：idx，_data,kd
///kd树使用方法：
///build(ll l,ll r,ll k)闭区间  k维度 输入为_data；
///query(kdnode p,ll m)在kd树里面找离p最近的m个点
///可能需要修改的地方
///函数kd_d 定义距离
///maxn
///数组xyz
/// if((p.xyz[dim] - data[rt].xyz[dim])*(p.xyz[dim] - data[rt].xyz[dim]) < Q.top().first)这里是距离

const int maxn=2e5+5;
#define lson rt << 1
#define rson rt << 1 | 1
#define Pair pair<double, kdnode>
//----------------------------------------------------------------------------------------
//kd树节点类型
ll idx;
struct kdnode
{
    ll xyz[5];     //定义属性数组
    bool operator < (const kdnode &u) const
    {
        return xyz[idx] < u.xyz[idx];
    }
} _data[maxn];  //_data[]数组代表输入的数据
//---------------------------------------------------------------------------/////////////

inline double kd_d(kdnode& a,kdnode& b)
{
    if(a.xyz[2]>b.xyz[2])
        return 9e18;
    else
    {
        double dx=a.xyz[0]-b.xyz[0];
        double dy=a.xyz[1]-b.xyz[1];
        return dx*dx+dy*dy+a.xyz[3]/3e5;
    }
}

//----------------------------------------------------------------------------------------
//kd树声明
class KDTree
{
public:
    void build(ll, ll, ll);             //闭区间建树
    void query(kdnode, ll);             //闭区间查询
    priority_queue<Pair> Q;             //队列Q用于存放离p最近的m个数据
    //这里本来准备用static的，但是用了就报错很烦不知道为什么。
    
private:
    ll k;
    kdnode data[4 * maxn];                //data[]数组代表K-D树的所有节点数据
    ll flag[4 * maxn];                    //用于标记某个节点是否存在，1表示存在，-1表示不存在
    void _build(ll,ll,ll,ll);             //查询
    void _query(kdnode, ll, ll, ll);      //查询
} kd;
//---------------------------------------------------------------------------/////////////


//----------------------------------------------------------------------------------------
//kd树实现
//查询函数，寻找离p最近的m个特征属性
void KDTree::query(kdnode P,ll m)
{
    //事先需要清空优先队列
    while(!Q.empty())
        Q.pop();
    _query(P,m,1,0);
}


void KDTree::build(ll l,ll r,ll kk)
{
    k=kk;
    _build(l,r,1,0);
}


//建树步骤，参数dept代表树的深度
void KDTree::_build(ll l, ll r, ll rt=1, ll dept=0)
{
    if(l > r)
        return;
    flag[rt] = 1;                   //表示编号为rt的节点存在
    flag[lson] = flag[rson] = -1;   //当前节点的孩子暂时标记不存在
    idx = dept % k;                 //按照编号为idx的属性进行划分
    ll mid = (l + r) >> 1;
    nth_element(_data + l, _data + mid, _data + r + 1);   //nth_element()为STL中的函数
    data[rt] = _data[mid];
    _build(l, mid - 1, lson, dept + 1);  //递归左子树
    _build(mid + 1, r, rson, dept + 1);  //递归右子树
}


//查询函数，寻找离p最近的m个特征属性
void KDTree::_query(kdnode p, ll m, ll rt=1, ll dept=0)
{
    if(flag[rt] == -1)
        return;   //不存在的节点不遍历
    Pair cur(0, data[rt]);       //获取当前节点的数据和到p的距离
    cur.first=kd_d(cur.second,p);//距离
    
    ll dim = dept % k;          //跟建树一样，这样能保证相同节点的dim值不变
    bool fg = 0;                 //用于标记是否需要遍历右子树
    ll x = lson;
    ll y = rson;
    if(p.xyz[dim] >= data[rt].xyz[dim]) //数据p的第dim个特征值大于等于当前的数据，则需要进入右子树
        swap(x, y);
    if(~flag[x])
        _query(p, m, x, dept + 1);      //如果节点x存在，则进入子树继续遍历
    
    //以下是回溯过程，维护一个优先队列
    if(Q.size() < m)   //如果队列没有满，则继续放入
    {
        Q.push(cur);
        fg = 1;
    }
    else
    {
        if(cur.first < Q.top().first)  //如果找到更小的距离，则用于替换队列Q中最大的距离的数据
        {
            Q.pop();
            Q.push(cur);
        }
        if((p.xyz[dim] - data[rt].xyz[dim])*(p.xyz[dim] - data[rt].xyz[dim]) < Q.top().first)
        {
            fg = 1;
        }
    }
    if(~flag[y] && fg)
        _query(p, m, y, dept + 1);
}
